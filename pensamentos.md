**Promessas**

```js
function fazerPromessas() {
  return new Promise((resolve, reject) => {
    var tcreq = new TCRequest();
    tcreq.open();
    tcreq.onload = function () {
      if (tcreq.status == 200) {
        resolve(tcreq.response);
      } else {
        reject(Error(""
          + "Não faça "
          + "promessas "
          + "se não pode "
          + "cumpri-las."
        ));
      }
    };
    tcreq.onerror = function () {
      reject(Error("Network Error"));
    };
    tcreq.send();
  });
}
```
